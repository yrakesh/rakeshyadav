SASS

SASS Mixin calling mixin by bypassing initial default value arguments

SASS has the concept of mixin which is just like function in other programming languages.
In SASS we can create mixins which have default values.
Below is the example:

	%yes_bg{
		background:url(path/to/img.png);
	}
	%no_bg{
		background:url(path/to/another_img.png);
	}
	@mixin my-mixin($color, $font-size:2em, $yes_bg:false){
		color:$color;
		font-size:$font-size;
		@if $yes_bg{
			@extend %yes_bg;
		}
		@else{
			@extend %no_bg;
		}
		.oldie &{
			@content;
		}
	}
	.class-no-bg{
		@include my-mixin(green);
	}
	.class-show-bg{
		@include my-mixin(blue, $yes_bg: true){
			box-shadow: 2px 2px 2px black;
			border:1px solid black;
		};
	}

Loops / iteration 


$color-names: alpha beta gamma;

@mixin color-iterator($color, $iterations, $hue, $names: $color-names){
  @for $i from 1 through $iterations{
    .#{nth($names, $i)}-color{
      $adjust: $i * $hue;
      $use-color: adjust_hue($color, $adjust);
      color:$use-color;
    }
  }
}
@include color-iterator(green, 3, 5)

 If / Else 


$ie-support: true;

@mixin if-demo($var){
	@if $ie-support {
		font-size: $var * 16px;
	}
	@if $var > 5 {
		font-size: $var + rem;
	}
	@for $i from 1 through 5{
		@if $i > 2 {
			output: $i;
		}
	}
}
div{
	@include if-demo( 7)	
}

$ie-support: false;
$var: 6;
div{
  font-size: if($ie-support, $var * 16px, $var + rem);
}
